bool witness(i64 a,i64 n)
{
  i64 x,d=1,i=ceil(log(n-1.0)/log(2.0))-1;
  for(;i>=0;i--)
  {
    x=d; d=(d*d)%n;
    if(d==1&&x!=1&&x!=n-1) return 1;
    if(((n-1)&(1<<i))>0) d=(d*a)%n;
  }
  return  d==1?0:1;
}
bool miller_rabin(i64 n)
{
  if(n==2)   return 1;
  if(n==1||(n&1)==0) return 0;
  i64 j,a;
  for(j=0;j<50;j++)
  {
   a=rand()*(n-2)/RAND_MAX+1;
   if(witness(a,n)) return 0;
  }
  return 1;
}

#include<string.h>
const int N = 1000005;
bool prim[N];
void isprim()
{
    memset(prim,true,sizeof(prim));
    for(int i=2;i*i<N;i++){
        if(prim[i]){
            for(int j=i*i;j<N;j=j+i){
                prim[j]=false;
            }
        }
    }
}

素荡虮恚
#define MAX_N 1000
int a[MAX_N+1];
int p[MAX_N+1];
int nCount=0;

void Init(int n) //线性筛法，不过在小范围上(约n<1e7)不比上一个方法快
{
    for (int i=2;i<=n;i++)
    {
        if (a[i]==0)
        {
            p[++nCount]=i;
        }
        for (int j=1,k; (j<=nCount) && (k=i*p[j])<=n; j++) //筛选循环
        {
            a[k] = 1 ;
            if (i%p[j] == 0) break;
        }
    }
}

#include <iostream>
#include <stdio.h>
int main(void)
{
    Init(MAX_N);
    for(int n=1; p[n]>1; ++n)
    {
        printf("%5d", p[n]);
    }
    return 0;
}
